백준 N과 M 문제집 중 중복순열 문제 코드

```java
import java.util.Scanner;

//1부터 N까지 서로 다른 N개의 자연수 중에서 중복을 허용하여 M개를 골라 나열한 중복순열 
//중복순열 N_Π_M를 오름차순으로 출력
public class Main_BJ_15651_3 {
	public static int N, M, idx[];
	public static StringBuilder sb;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		sb = new StringBuilder();

		N = sc.nextInt();
		M = sc.nextInt();

		idx = new int[M]; //
		perm(0);

		System.out.print(sb);
	}

	private static void perm(int depth) {
		if (depth == M) {
			for (int m = 0; m < M; m++) {
				sb.append(idx[m]).append(' ');
			}
			sb.append('\n');
			return;
		}
		for (int i = 1; i <= N; i++) {
			idx[depth] = i;
			perm(depth + 1);
		}
	}
}
```

```java
import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.Arrays;
import java.util.StringTokenizer;

//n[0]부터 n[N-1]까지 N개의 자연수 중에서 중복을 허용하여 M개를 골라 나열한 중복순열 
//중복순열 N_Π_M를 오름차순으로 출력 (<-n을 정렬)

//n에 값이 중복으로 들어올 수 있음!!!! 
//같은 자리에 같은 값이 들어오는 경우는 모두 동일한 경우이다.
public class Main_BJ_15665_11 {
	public static StringTokenizer st;
	public static StringBuilder sb;
	public static int N, M, n[], idx[];
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

		st = new StringTokenizer(br.readLine(), " ");
		N = Integer.parseInt(st.nextToken());
		M = Integer.parseInt(st.nextToken());

		n = new int[N];
		st = new StringTokenizer(br.readLine(), " ");
		int x = 0;
		while (st.hasMoreTokens()) {
			n[x++] = Integer.parseInt(st.nextToken());
		}
		Arrays.sort(n);

		idx = new int[M];
		sb = new StringBuilder();
		perm(0);

		bw.write(sb.toString());
		bw.flush();
	}
	
	private static void perm(int depth) {
		if(depth == M) {
			for(int m=0; m<M; m++) {
				sb.append(idx[m]).append(' ');
			}
			sb.append('\n');
			return;
		}
		for(int i=0; i<N; i++) {
			if(idx[depth] != n[i]) { ////idx의 어떤 자리에 같은 수가 왔던 적이 있는지 체크
				idx[depth] = n[i]; //
				perm(depth+1);
			}
		}
		idx[depth] = 0; ////자연수가 아닌 수로 초기화
	}
}
```
