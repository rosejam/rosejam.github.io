package N과M;

import java.util.Scanner;

//1부터 N까지 서로 다른 N개의 자연수 중에서 중복 없이 M개를 고른 조합 
//조합 N_C_M를 오름차순으로 출력
public class Main_BJ_15650_2 {
	public static int N, M, idx[];
	public static StringBuilder sb;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		sb = new StringBuilder();

		N = sc.nextInt();
		M = sc.nextInt();

		idx = new int[M]; //
		comb(0, 1);

		System.out.print(sb);
	}

	private static void comb(int depth, int start) {
		if (depth == M) {
			for (int m = 0; m < M; m++) {
				sb.append(idx[m]).append(' ');
			}
			sb.append('\n');
			return;
		}
		for (int i = start; i <= N; i++) {
			idx[depth] = i;
			comb(depth + 1, i + 1); // i가 아닌 i+1
		}
	}
}

package N과M;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.Arrays;
import java.util.StringTokenizer;

//n[0]부터 n[N-1]까지 N개의 자연수 중에서 중복 없이 M개를 고른 조합 
//조합 N_C_M를 오름차순으로 출력 (<-n을 정렬)

//n에 값이 중복으로 들어올 수 있음!!!! 
//같은 자리에 같은 값이 들어오는 경우는 모두 동일한 경우이다.
public class Main_BJ_15664_10 {
	public static StringTokenizer st;
	public static StringBuilder sb;
	public static int N, M, n[], idx[];

	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

		st = new StringTokenizer(br.readLine(), " ");
		N = Integer.parseInt(st.nextToken());
		M = Integer.parseInt(st.nextToken());

		n = new int[N];
		st = new StringTokenizer(br.readLine(), " ");
		int x = 0;
		while (st.hasMoreTokens()) {
			n[x++] = Integer.parseInt(st.nextToken());
		}
		Arrays.sort(n);

		idx = new int[M];
		sb = new StringBuilder();
		comb(0, 0);

		bw.write(sb.toString());
		bw.flush();
	}

	private static void comb(int depth, int start) {
		if (depth == M) {
			for (int m = 0; m < M; m++) {
				sb.append(idx[m]).append(' ');
			}
			sb.append('\n');
			return;
		}
		for (int i = start; i < N; i++) {
			if (idx[depth] != n[i]) { //// idx의 어떤 자리에 같은 수가 왔던 적이 있는지 체크
				idx[depth] = n[i]; //
				comb(depth + 1, i + 1); // i가 아닌 i+1
			}
		}
		idx[depth] = 0; //// 자연수가 아닌 수로 초기화
	}
}
